193 void PHActsTrkFitter::updateSvtxTrack(const Acts::KalmanFitterResult<SourceLink>& fitOutput, const unsigned int trackKey)
194 {
195  const auto& params = fitOutput.fittedParameters.value();
196 
197  SvtxTrackMap::Iter trackIter = m_trackMap->find(trackKey);
198  SvtxTrack *track = trackIter->second;
199 
201  track->set_x(params.position()(0) / Acts::UnitConstants::cm);
202  track->set_y(params.position()(1) / Acts::UnitConstants::cm);
203  track->set_z(params.position()(2) / Acts::UnitConstants::cm);
204  track->set_px(params.momentum()(0));
205  track->set_py(params.momentum()(1));
206  track->set_pz(params.momentum()(2));
207 
209  Acts::MultiTrajectory<SourceLink> states = fitOutput.fittedStates;
210 
211  if(params.covariance())
212  {
213  ActsCovarianceRotater *rotater = new ActsCovarianceRotater();
214  Acts::BoundSymMatrix rotatedCov =
215  rotater->rotateActsCovToSvtxTrack(fitOutput);
216 
217  for(int i = 0; i < 6; i++)
218  {
219  for(int j = 0; j < 6; j++)
220  {
221  track->set_error(i,j, rotatedCov(i,j));
222  }
223  }
224  }
225 
226  return;
