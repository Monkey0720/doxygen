46 FW::FittingAlgorithm::FitterFunction FW::FittingAlgorithm::makeFitterFunction(
47  std::shared_ptr<const Acts::TrackingGeometry> trackingGeometry,
48  Options::BFieldVariant magneticField, Acts::Logging::Level lvl) {
49  using Updater = Acts::GainMatrixUpdater<Acts::BoundParameters>;
50  using Smoother = Acts::GainMatrixSmoother<Acts::BoundParameters>;
51 
52  // unpack the magnetic field variant and instantiate the corresponding fitter.
53  return std::visit(
54  [trackingGeometry, lvl](auto&& inputField) -> FitterFunction {
55  // each entry in the variant is already a shared_ptr
56  // need ::element_type to get the real magnetic field type
57  using InputMagneticField =
58  typename std::decay_t<decltype(inputField)>::element_type;
59  using MagneticField = Acts::SharedBField<InputMagneticField>;
60  using Stepper = Acts::EigenStepper<MagneticField>;
61  using Navigator = Acts::Navigator;
62  using Propagator = Acts::Propagator<Stepper, Navigator>;
63  using Fitter = Acts::KalmanFitter<Propagator, Updater, Smoother>;
64 
65  // construct all components for the fitter
66  MagneticField field(std::move(inputField));
67  Stepper stepper(std::move(field));
68  Navigator navigator(trackingGeometry);
69  navigator.resolvePassive = false;
70  navigator.resolveMaterial = true;
71  navigator.resolveSensitive = true;
72  Propagator propagator(std::move(stepper), std::move(navigator));
73  Fitter fitter(std::move(propagator),
74  Acts::getDefaultLogger("KalmanFitter", lvl));
75 
76  // build the fitter functions. owns the fitter object.
77  return FitterFunctionImpl<Fitter>(std::move(fitter));
78  },
79  std::move(magneticField));
80 }
