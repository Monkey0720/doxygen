117 void TpcSpaceChargeReconstruction::process_track( SvtxTrack* track )
118 {
119 
120  // running track state
121  auto state_iter = track->begin_states();
122 
123  // loop over clusters
124  for( auto key_iter = track->begin_cluster_keys(); key_iter != track->end_cluster_keys(); ++key_iter )
125  {
126 
127  const auto& cluster_key = *key_iter;
128  auto cluster = m_cluster_map->findCluster( cluster_key );
129  if( !cluster )
130  {
131  std::cout << PHWHERE << " unable to find cluster for key " << cluster_key << std::endl;
132  continue;
133  }
134 
135  // should make sure that cluster belongs to TPC
136  const auto layer = TrkrDefs::getLayer(cluster_key);
137  if( layer < m_firstlayer_tpc || layer >= m_firstlayer_tpc + m_nlayers_tpc ) continue;
138 
139  // cluster r, phi and z
140  const auto cluster_r = get_r( cluster->getX(), cluster->getY() );
141  const auto cluster_phi = std::atan2( cluster->getY(), cluster->getX() );
142  const auto cluster_z = cluster->getZ();
143 
144  // cluster errors
145  const auto cluster_rphi_error = cluster->getRPhiError();
146  const auto cluster_z_error = cluster->getZError();
147 
148  /*
149  remove clusters with too small errors since they are likely pathological
150  and have a large contribution to the chisquare
151  TODO: make these cuts configurable
152  */
153  if( cluster_rphi_error < 0.015 ) continue;
154  if( cluster_z_error < 0.05 ) continue;
155 
156  // find track state that is the closest to cluster
157  /* this assumes that both clusters and states are sorted along r */
158  float dr_min = -1;
159  for( auto iter = state_iter; iter != track->end_states(); ++iter )
160  {
161  const auto dr = std::abs( cluster_r - get_r( iter->second->get_x(), iter->second->get_y() ) );
162  if( dr_min < 0 || dr < dr_min )
163  {
164  state_iter = iter;
165  dr_min = dr;
166  } else break;
167  }
168 
169  // get relevant track state
170  const auto state = state_iter->second;
171 
172  // track errors
173  const auto track_rphi_error = state->get_rphi_error();
174  const auto track_z_error = state->get_z_error();
175 
176  // also cut on track errors
177  // warning: smaller errors are probably needed when including outer tracker
178  if( track_rphi_error < 0.015 ) continue;
179  if( track_z_error < 0.1 ) continue;
180 
181  // extrapolate track parameters to the cluster r
182  const auto track_r = get_r( state->get_x(), state->get_y() );
183  const auto dr = cluster_r - track_r;
184  const auto track_drdt = get_r( state->get_px(), state->get_py() );
185  const auto track_dxdr = state->get_px()/track_drdt;
186  const auto track_dydr = state->get_py()/track_drdt;
187  const auto track_dzdr = state->get_pz()/track_drdt;
188 
189  // store state position
190  const auto track_x = state->get_x() + dr*track_dxdr;
191  const auto track_y = state->get_y() + dr*track_dydr;
192  const auto track_z = state->get_z() + dr*track_dzdr;
193  const auto track_phi = std::atan2( track_y, track_x );
194 
195  // get residual errors squared
196  const auto erp = square(track_rphi_error) + square(cluster_rphi_error);
197  const auto ez = square(track_z_error) + square(cluster_z_error);
198 
199  // sanity check
200  if( std::isnan( erp ) ) continue;
201  if( std::isnan( ez ) ) continue;
202 
203  // get residuals
204  const auto drp = cluster_r*delta_phi( track_phi - cluster_phi );
205  const auto dz = track_z - cluster_z;
206 
207  // get track angles
208  const auto cosphi( std::cos( track_phi ) );
209  const auto sinphi( std::sin( track_phi ) );
210  const auto track_pphi = -state->get_px()*sinphi + state->get_py()*cosphi;
211  const auto track_pr = state->get_px()*cosphi + state->get_py()*sinphi;
212  const auto track_pz = state->get_pz();
213  const auto talpha = -track_pphi/track_pr;
214  const auto tbeta = -track_pz/track_pr;
215 
216  // check against limits
217  // TODO: make this configurable
218  static constexpr float max_talpha = 0.6;
219  static constexpr float max_residual = 5;
220  if( std::abs( talpha ) > max_talpha ) continue;
221  if( std::abs( drp ) > max_residual ) continue;
222 
223  // get cell
224  const auto i = get_cell( cluster_key, cluster );
225 
226  if( i < 0 || i >= m_totalbins ) continue;
227 
228  m_lhs[i](0,0) += 1./erp;
229  m_lhs[i](0,1) += 0;
230  m_lhs[i](0,2) += talpha/erp;
231 
232  m_lhs[i](1,0) += 0;
233  m_lhs[i](1,1) += 1./ez;
234  m_lhs[i](1,2) += tbeta/ez;
235 
236  m_lhs[i](2,0) += talpha/erp;
237  m_lhs[i](2,1) += tbeta/ez;
238  m_lhs[i](2,2) += square(talpha)/erp + square(tbeta)/ez;
239 
240  m_rhs[i](0,0) += drp/erp;
241  m_rhs[i](1,0) += dz/ez;
242  m_rhs[i](2,0) += talpha*drp/erp + tbeta*dz/ez;
243 
244  ++m_cluster_count[i];
245 
246  }

251 void TpcSpaceChargeReconstruction::calculate_distortions( PHCompositeNode* topNode )
252 {
253 
254  // get tpc geometry
255  auto *geom_container = findNode::getClass<PHG4CylinderCellGeomContainer>(topNode, "CYLINDERCELLGEOM_SVTX");
256  if (!geom_container)
257  {
258  std::cout << PHWHERE << " can't find node CYLINDERCELLGEOM_SVTX" << std::endl;
259  return;
260  }
261 
262  // calculate distortions in each volume elements
263  std::vector<column_t> delta(m_totalbins);
264  std::vector<matrix_t> cov(m_totalbins);
265 
266  for( int i = 0; i < m_totalbins; ++i )
267  {
268  cov[i] = m_lhs[i].inverse();
269  delta[i] = m_lhs[i].partialPivLu().solve( m_rhs[i] );
270  }
271 
272  // create tgraphs
273  using TGraphPointer = std::unique_ptr<TGraphErrors>;
274  std::vector<TGraphPointer> tg( m_zbins*m_phibins*m_ncoord );
275 
276  for( int iz = 0; iz < m_zbins; ++iz )
277  for( int iphi = 0; iphi < m_phibins; ++iphi )
278  for( int icoord = 0; icoord < m_ncoord; ++icoord )
279  {
280 
281  const int tgindex = iz + m_zbins*( iphi + m_phibins*icoord );
282  tg[tgindex].reset( new TGraphErrors() );
283  tg[tgindex]->SetName( Form( "tg_%i_%i_%i", iz, iphi, icoord ) );
284 
285  for( int ir = 0; ir < m_rbins; ++ir )
286  {
287 
288  // get layers corresponding to bins
289  const int inner_layer = m_firstlayer_tpc + m_nlayers_tpc*ir/m_rbins;
290  const int outer_layer = m_firstlayer_tpc + m_nlayers_tpc*(ir+1)/m_rbins-1;
291 
292  const auto inner_radius = geom_container->GetLayerCellGeom(inner_layer)->get_radius();
293  const auto outer_radius = geom_container->GetLayerCellGeom(outer_layer)->get_radius();
294  const float r = (inner_radius+outer_radius)/2;
295 
296  int index = get_cell( iz, ir, iphi );
297  tg[tgindex]->SetPoint( ir, r, delta[index](icoord,0) );
298  tg[tgindex]->SetPointError( ir, 0, std::sqrt(cov[index](icoord,icoord)) );
299  }
300  }
301 
302  // save to root file
303  {
304  std::unique_ptr<TFile> outputfile( TFile::Open( m_outputfile.c_str(), "RECREATE" ) );
305  outputfile->cd();
306  for( auto&& value:tg ) { value->Write(); }
307  outputfile->Close();
308  }
309 }

321 int TpcSpaceChargeReconstruction::get_cell( TrkrDefs::cluskey cluster_key, TrkrCluster* cluster ) const
322 {
323  // radius
324  const auto layer = TrkrDefs::getLayer(cluster_key);
325  const int ir = m_rbins*(layer - m_firstlayer_tpc)/m_nlayers_tpc;
326 
327  // azimuth
328  auto cluster_phi = std::atan2( cluster->getY(), cluster->getX() );
329  if( cluster_phi >= M_PI ) cluster_phi -= 2*M_PI;
330  const int iphi = m_phibins*(cluster_phi + M_PI)/(2.*M_PI);
331 
332  // z
333  // TODO: get TPC dimension from recoconst ?
334  const auto cluster_z = cluster->getZ();
335  static constexpr float z_min = -212/2;
336  static constexpr float z_max = 212/2;
337  const int iz = m_zbins*(cluster_z-z_min)/(z_max-z_min);
338 
339  return get_cell( iz, ir, iphi );
