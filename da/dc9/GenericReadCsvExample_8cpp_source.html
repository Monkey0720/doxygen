23 int main(int argc, char* argv[]) {
24  GenericDetector detector;
25 
26  // setup and parse options
27  auto desc = FW::Options::makeDefaultOptions();
28  FW::Options::addSequencerOptions(desc);
29  FW::Options::addGeometryOptions(desc);
30  FW::Options::addMaterialOptions(desc);
31  FW::Options::addInputOptions(desc);
32  detector.addOptions(desc);
33 
34  auto vm = FW::Options::parse(desc, argc, argv);
35  if (vm.empty()) {
36  return EXIT_FAILURE;
37  }
38 
39  FW::Sequencer sequencer(FW::Options::readSequencerConfig(vm));
40 
41  // Read some standard options
42  auto logLevel = FW::Options::readLogLevel(vm);
43  auto inputDir = vm["input-dir"].as<std::string>();
44 
45  // Setup detector geometry
46  auto geometry = FW::Geometry::build(vm, detector);
47  auto trackingGeometry = geometry.first;
48  // Add context decorators
49  for (auto cdr : geometry.second) {
50  sequencer.addContextDecorator(cdr);
51  }
52 
53  // Read particles from CSV files
54  auto particleReaderCfg = FW::Options::readCsvParticleReaderConfig(vm);
55  particleReaderCfg.outputParticles = "particles";
56  sequencer.addReader(
57  std::make_shared<FW::CsvParticleReader>(particleReaderCfg, logLevel));
58 
59  // Read clusters from CSV files
60  auto clusterReaderCfg = FW::Options::readCsvPlanarClusterReaderConfig(vm);
61  clusterReaderCfg.trackingGeometry = trackingGeometry;
62  clusterReaderCfg.outputClusters = "clusters";
63  clusterReaderCfg.outputHitParticlesMap = "hit_particle_map";
64  clusterReaderCfg.outputHitIds = "hit_ids";
65  sequencer.addReader(
66  std::make_shared<FW::CsvPlanarClusterReader>(clusterReaderCfg, logLevel));
67 
68  // Print some information as crosscheck
69  FW::PrintHits::Config printCfg;
70  printCfg.inputClusters = clusterReaderCfg.outputClusters;
71  printCfg.inputHitParticlesMap = clusterReaderCfg.outputHitParticlesMap;
72  printCfg.inputHitIds = clusterReaderCfg.outputHitIds;
73  // the following print selections work for the original author.
74  // you probably need to adapt them to your data.
75  printCfg.hitIdStart = 10224;
76  printCfg.hitIdLength = 8;
77  printCfg.volumeId = 13;
78  printCfg.layerId = 4;
79  printCfg.moduleId = 116;
80  sequencer.addAlgorithm(std::make_shared<FW::PrintHits>(printCfg, logLevel));
81 
82  return sequencer.run();
83 }
