205 int make_LANL_FST_station(string name, PHG4Reco *g4Reco, double zpos, double Rmin,
206  double Rmax)
207 {
208  // cout
209  // << "make_GEM_station - GEM construction with PHG4SectorSubsystem - make_GEM_station_EdgeReadout of "
210  // << name << endl;
211 
212  // always facing the interaction point
213  double polar_angle = 0;
214  if (zpos < 0)
215  {
216  zpos = -zpos;
217  polar_angle = TMath::Pi();
218  }
219 
220  double min_polar_angle = TMath::ATan2(Rmin, zpos);
221  double max_polar_angle = TMath::ATan2(Rmax, zpos);
222 
223  if (max_polar_angle < min_polar_angle)
224  {
225  double t = max_polar_angle;
226  max_polar_angle = min_polar_angle;
227  min_polar_angle = t;
228  }
229 
230  PHG4SectorSubsystem *fst;
231  fst = new PHG4SectorSubsystem(name.c_str());
232 
233  fst->SuperDetector(name);
234 
235  fst->get_geometry().set_normal_polar_angle(polar_angle);
236  fst->get_geometry().set_normal_start(
237  zpos * PHG4Sector::Sector_Geometry::Unit_cm());
238  fst->get_geometry().set_min_polar_angle(min_polar_angle);
239  fst->get_geometry().set_max_polar_angle(max_polar_angle);
240  fst->get_geometry().set_max_polar_edge(
241  PHG4Sector::Sector_Geometry::ConeEdge());
242  fst->get_geometry().set_min_polar_edge(
243  PHG4Sector::Sector_Geometry::ConeEdge());
244  fst->get_geometry().set_N_Sector(1);
245  fst->get_geometry().set_material("G4_AIR");
246  fst->OverlapCheck(overlapcheck);
247 
248  const double cm = PHG4Sector::Sector_Geometry::Unit_cm();
249  const double mm = .1 * cm;
250  const double um = 1e-3 * mm;
251  // build up layers
252  fst->get_geometry().AddLayer("SliconSensor", "G4_Si", 18 * um, true, 100);
253  fst->get_geometry().AddLayer("Metalconnection", "G4_Al", 15 * um, false, 100);
254  fst->get_geometry().AddLayer("SliconSupport", "G4_Al", 285 * um, false, 100);
255  fst->get_geometry().AddLayer("HDI", "G4_KAPTON", 20 * um, false, 100);
256  fst->get_geometry().AddLayer("Cooling", "G4_WATER", 100 * um, false, 100);
257  fst->get_geometry().AddLayer("Support", "G4_GRAPHITE", 50 * um, false, 100);
258  fst->get_geometry().AddLayer("Support_Gap", "G4_AIR", 1 * cm, false, 100);
259  fst->get_geometry().AddLayer("Support2", "G4_GRAPHITE", 50 * um, false, 100);
260 
261  g4Reco->registerSubsystem(fst);
262  return 0;
263 }
