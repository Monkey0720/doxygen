82 void Detector::FillEventKinematics(Event* eventS) {
83  if (!(useNM || useJB || useDA)) {
84  return;
85  } // if
86  // Need a bit of jiggery-pokery here, as the incident beam info isn't
87  // associated with the smeared event.
88  // So, get the beam info from the MC event, but replace the scattered
89  // electron with the smeared version.
90  // Then we can use the standard JB/DA algorithms on the smeared event.
91  const ParticleMCS* scattered = eventS->ScatteredLepton();
92  typedef std::unique_ptr<erhic::DisKinematics> KinPtr;
93  if (useNM && scattered) {
94  KinPtr kin(erhic::LeptonKinematicsComputer(*eventS).Calculate());
95  if (kin.get()) {
96  eventS->SetLeptonKinematics(*kin);
97  } // if
98  } else {
99  eventS->SetLeptonKinematics( erhic::DisKinematics(-1., -1., -1., -1., -1.));
100  } // if
101  if (useJB) {
102  KinPtr kin(erhic::JacquetBlondelComputer(*eventS).Calculate());
103  if (kin.get()) {
104  eventS->SetJacquetBlondelKinematics(*kin);
105  } // if
106  } // if
107  if (useDA && scattered) {
108  KinPtr kin(erhic::DoubleAngleComputer(*eventS).Calculate());
109  if (kin.get()) {
110  eventS->SetDoubleAngleKinematics(*kin);
111  } // if
112  } // if
113 }

116  std::list<Smearer*> devices;

131 ParticleMCS* Detector::Smear(const erhic::VirtualParticle& prt) const {
132  // Does the particle fall in the acceptance of any device?
133  // If so, we smear it, if not, we skip it (store a NULL pointer).
134  std::list<Smearer*> devices = Accept(prt);
135  ParticleMCS* prtOut(NULL);
136  if (!devices.empty()) {
137  // It passes through at least one device, so smear it.
138  // Devices in which it doesn't pass won't smear it.
139  prtOut = new ParticleMCS();
140  std::list<Smearer*>::iterator iter;
141  for (iter = devices.begin(); iter != devices.end(); ++iter) {
142  (*iter)->Smear(prt, *prtOut);
143  } // for
144  // Compute derived momentum components.
145  prtOut->px = prtOut->p * sin(prtOut->theta) * cos(prtOut->phi);
146  prtOut->py = prtOut->p * sin(prtOut->theta) * sin(prtOut->phi);
147  prtOut->pt = sqrt(pow(prtOut->px, 2.) + pow(prtOut->py, 2.));
148  prtOut->pz = prtOut->p * cos(prtOut->theta);
149  } // if
150  return prtOut;
151 }
