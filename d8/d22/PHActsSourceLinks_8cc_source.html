256 Surface PHActsSourceLinks::getTpcLocalCoords(Acts::Vector2D &local2D,
257  Acts::BoundMatrix &localErr,
258  const TrkrCluster *cluster,
259  const TrkrDefs::cluskey clusKey)
260 {
261  // cm
262  const float x = cluster->getPosition(0);
263  const float y = cluster->getPosition(1);
264  const float z = cluster->getPosition(2);
265 
266  // In local coords the covariances are in the r*phi vs z frame
267  // They have been rotated into global coordinates in TrkrCluster
268  TMatrixD worldErr(3, 3);
269  for (int i = 0; i < 3; ++i)
270  {
271  for (int j = 0; j < 3; j++)
272  {
273  worldErr[i][j] = cluster->getError(i, j); // this is the cov error squared
274  }
275  }
276 
278  TVector3 world(x, y, z);
279 
281  const double clusPhi = atan2(world[1], world[0]);
282  const double radius = sqrt(x * x + y * y) * Acts::UnitConstants::cm;
283  const double rClusPhi = radius * clusPhi;
284  const double zTpc = world[2] * Acts::UnitConstants::cm;
285 
286  const unsigned int layer = TrkrDefs::getLayer(clusKey);
287  const unsigned int sectorId = TpcDefs::getSectorId(clusKey);
288  const unsigned int side = TpcDefs::getSide(clusKey);
289 
291  TrkrDefs::hitsetkey tpcHitSetKey = TpcDefs::genHitSetKey(layer, sectorId, side);
292  std::vector<double> worldVec = {world[0], world[1], world[2]};
295  Surface surface = m_actsGeometry->getTpcSurfaceFromCoords(tpcHitSetKey,
296  worldVec);
297  assert(surface);
298 
299  if(Verbosity() > 0)
300  {
301  std::cout << "Stream of found TPC surface: " << std::endl;
302  surface->toStream(m_tGeometry->geoContext, std::cout);
303  }
304 
308  // lengths in mm
309  Acts::Vector3D center = surface->center(m_actsGeometry->getGeoContext());
310  Acts::Vector3D normal = surface->normal(m_actsGeometry->getGeoContext());
311 
312  double surfRadius = sqrt(center[0]*center[0] + center[1]*center[1]);
313  double surfPhiCenter = atan2(center[1], center[0]);
314  double surfRphiCenter = atan2(center[1], center[0]) * surfRadius;
315  double surfZCenter = center[2];
316 
317  if (Verbosity() > 0)
318  {
319  std::cout << std::endl << "surface center readback: x " << center[0]
320  << " y " << center[1] << " z " << center[2] << " radius " << surfRadius << std::endl;
321  std::cout << "Surface normal vector : "<< normal(0) << ", "
322  << normal(1) << ", " << normal(2) << std::endl;
323  std::cout << " surface center phi " << atan2(center[1], center[0])
324  << " surface center r*phi " << surfRphiCenter
325  << " surface center z " << surfZCenter
326  << std::endl;
327  }
328 
329  Acts::Vector3D globalPos(x * Acts::UnitConstants::cm,
330  y * Acts::UnitConstants::cm,
331  z * Acts::UnitConstants::cm);
332 
333  surface->globalToLocal(m_actsGeometry->getGeoContext(), globalPos,
334  surface->normal(m_actsGeometry->getGeoContext()),
335  local2D);
336 
337  if (Verbosity() > 0)
338  {
339  std::cout << "cluster readback (mm): x " << x*Acts::UnitConstants::cm << " y " << y*Acts::UnitConstants::cm << " z " << z*Acts::UnitConstants::cm
340  << " radius " << radius << std::endl;
341  std::cout << " cluster phi " << clusPhi << " cluster z " << zTpc << " r*clusphi " << rClusPhi << std::endl;
342  std::cout << " local phi " << clusPhi - surfPhiCenter
343  << " local rphi " << rClusPhi-surfRphiCenter
344  << " local z " << zTpc - surfZCenter << std::endl;
345  std::cout << " acts local : " <<local2D(0) <<" "<<local2D(1) << std::endl;
346  }
347 
348  TMatrixD sPhenixLocalErr = transformCovarToLocal(clusPhi, worldErr);
349 
351  localErr(Acts::eLOC_0, Acts::eLOC_0) =
352  sPhenixLocalErr[1][1] * Acts::UnitConstants::cm2;
353  localErr(Acts::eLOC_1, Acts::eLOC_0) =
354  sPhenixLocalErr[2][1] * Acts::UnitConstants::cm2;
355  localErr(Acts::eLOC_0, Acts::eLOC_1) =
356  sPhenixLocalErr[1][2] * Acts::UnitConstants::cm2;
357  localErr(Acts::eLOC_1, Acts::eLOC_1) =
358  sPhenixLocalErr[2][2] * Acts::UnitConstants::cm2;
359 
360 
361  if(Verbosity() > 0)
362  {
363  for (int i = 0; i < 3; ++i)
364  {
365  for (int j = 0; j < 3; j++)
366  {
367  std::cout << " " << i << " " << j << " worldErr " << worldErr[i][j] << " localErr " << sPhenixLocalErr[i][j] << std::endl;
368  }
369  }
370  }
371 
372  return surface;
373 }

828 TMatrixD PHActsSourceLinks::transformCovarToLocal(const double ladderPhi,
829  TMatrixD worldErr)
830 {
831  TMatrixD localErr(3, 3);
832 
833  // this is the matrix that was used to rotate from local to global coords
834  TMatrixD ROT(3, 3);
835  ROT[0][0] = cos(ladderPhi);
836  ROT[0][1] = -1.0 * sin(ladderPhi);
837  ROT[0][2] = 0.0;
838  ROT[1][0] = sin(ladderPhi);
839  ROT[1][1] = cos(ladderPhi);
840  ROT[1][2] = 0.0;
841  ROT[2][0] = 0.0;
842  ROT[2][1] = 0.0;
843  ROT[2][2] = 1.0;
844  // we want the inverse rotation
845  ROT.Invert();
846 
847  TMatrixD ROT_T(3, 3);
848  ROT_T.Transpose(ROT);
849 
850  localErr = ROT * worldErr * ROT_T;
851 
852  return localErr;
853 }
