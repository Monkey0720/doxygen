316 void PHCASeeding::FillTree()
317 {
318  // bgi::rtree<pointKey, bgi::quadratic<16> > rtree;
319  PHTimer *t_fill = new PHTimer("t_fill");
320  t_fill->stop();
321  int n_dupli = 0;
322  int nlayer[60];
323  for (int j = 0; j < 60; ++j) nlayer[j] = 0;
324 
325  TrkrClusterContainer::ConstRange clusrange = _cluster_map->getClusters();
326 
327  for (TrkrClusterContainer::ConstIterator iter = clusrange.first; iter != clusrange.second; ++iter)
328  {
329  TrkrCluster *cluster = iter->second;
330  TrkrDefs::cluskey ckey = iter->first;
331  unsigned int layer = TrkrDefs::getLayer(ckey);
332  if (layer < (_nlayers_maps + _nlayers_intt + _nlayers_tpc*2./3.)) continue;
333 
334  TVector3 vec(cluster->getPosition(0) - _vertex->get_x(), cluster->getPosition(1) - _vertex->get_y(), cluster->getPosition(2) - _vertex->get_z());
335 
336  double clus_phi = vec.Phi();
337  clus_phi -= 2 * M_PI * floor(clus_phi / (2 * M_PI));
338  double clus_eta = vec.Eta();
339  double clus_l = layer; // _radii_all[layer];
340 
341  vector<pointKey> testduplicate;
342  QueryTree(_rtree, clus_phi - 0.00001, clus_eta - 0.00001, layer - 0.5, clus_phi + 0.00001, clus_eta + 0.00001, layer + 0.5, testduplicate);
343  if (!testduplicate.empty())
344  {
345  ++n_dupli;
346  continue;
347  }
348  ++nlayer[layer];
349  t_fill->restart();
350  _rtree.insert(std::make_pair(point(clus_phi, clus_eta, clus_l), ckey));
351  t_fill->stop();
352  }
353 
354  std::cout << "fill time: " << t_fill->get_accumulated_time() / 1000. << " sec" << std::endl;
355  std::cout << "number of duplicates : " << n_dupli << std::endl;
356 }
