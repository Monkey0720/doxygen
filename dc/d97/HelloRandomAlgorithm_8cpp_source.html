28 FW::ProcessCode FW::HelloRandomAlgorithm::execute(
29  const AlgorithmContext& ctx) const {

32  // Create the local random number generator
33  FW::RandomEngine rng = m_cfg.randomNumbers->spawnGenerator(ctx);

36  std::normal_distribution<double> gaussDist(m_cfg.gaussParameters[0],

39  m_cfg.uniformParameters[0], m_cfg.uniformParameters[1]);
40  std::gamma_distribution<double> gammaDist(m_cfg.gammaParameters[0],
41  m_cfg.gammaParameters[1]);
42  std::poisson_distribution<int> poissonDist(m_cfg.poissonParameter);
43 
44  ACTS_INFO(m_cfg.drawsPerEvent << " draws per event will be done");
45 
46  // generate collection of random numbers
47  HelloDataCollection collection;
48  for (size_t idraw = 0; idraw < m_cfg.drawsPerEvent; ++idraw) {
49  double gauss = gaussDist(rng);
50  double uniform = uniformDist(rng);
51  double gamma = gammaDist(rng);
52  int poisson = poissonDist(rng);
53 
54  ACTS_VERBOSE("Gauss : " << gauss);
55  ACTS_VERBOSE("Uniform : " << uniform);
56  ACTS_VERBOSE("Gamma : " << gamma);
57  ACTS_VERBOSE("Poisson : " << poisson);
58 
59  HelloData x;

61  x.a = uniform;
62  x.b = gamma;
63  x.t = poisson;
64  collection.push_back(x);
65  }
66 
67  // transfer generated data to the event store.
68  ctx.eventStore.add(m_cfg.output, std::move(collection));
69 
70  return FW::ProcessCode::SUCCESS;
