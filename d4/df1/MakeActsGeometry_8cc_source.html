240 
241  if(m_verbosity > 30)

244  << box_r_phi << " ref arc " << m_surfStepPhi*m_layerRadius[ilayer] << " dz " << m_surfStepZ << endl;
245  tpc_gas_measurement_vol[ilayer]->Print();
246  }
247  }

249  int copy = 0;

256  {
257  for (unsigned int iphi = 0; iphi < m_nSurfPhi; ++iphi)

259

262  double phi_center_degrees = phi_center * 180.0 / M_PI;
263 
264  for (unsigned int ilayer = 0; ilayer < m_nTpcLayers; ++ilayer)
265  {
266  copy++;
267 
268  // place copies of the gas volume to fill up the layer
269 
270  double x_center = m_layerRadius[ilayer] * cos(phi_center);
271  double y_center = m_layerRadius[ilayer] * sin(phi_center);
272 
273  char rot_name[500];

279 
280  if(m_verbosity > 30 && ilayer == 30)

283  cout << " x_center " << x_center << " y_center " << y_center << " z_center " << z_center << " phi_center_degrees " << phi_center_degrees << endl;

693 TrkrDefs::hitsetkey MakeActsGeometry::getTpcHitSetKeyFromCoords(std::vector<double> &world)
694 {
695  // Look up TPC surface index values from world position of surface center
696  // layer
697  unsigned int layer = 999;
698  double layer_rad = sqrt(pow(world[0],2) + pow(world[1],2));
699  for(unsigned int ilayer=0;ilayer<m_nTpcLayers;++ilayer)
700  {
701  double tpc_ref_radius_low = m_layerRadius[ilayer] - m_layerThickness[ilayer] / 2.0;
702  double tpc_ref_radius_high = m_layerRadius[ilayer] + m_layerThickness[ilayer] / 2.0;
703  if(layer_rad >= tpc_ref_radius_low && layer_rad < tpc_ref_radius_high)
704  {
705  layer = ilayer;
706  break;
707  }
708  }
709  if(layer >= m_nTpcLayers)
710  {
711  cout << PHWHERE << "Error: undefined layer, do nothing world = " << world[0] << " " << world[1] << " " << world[2] << " layer " << layer << endl;
712  return Fun4AllReturnCodes::ABORTEVENT;
713  }
714  layer += 7;
715 
716  // side - from world z sign
717  unsigned int side;
718  if(world[2] < 0)
719  side = 0;
720  else
721  side = 1;
722 
723  // readout module
724  unsigned int readout_mod = 999;
725  double phi_world = atan2(world[1], world[0]);
726  for(unsigned int imod=0; imod<m_nTpcModulesPerLayer; ++imod)
727  {
728  double min_phi = m_modulePhiStart + (double) imod * m_moduleStepPhi;
729  double max_phi = m_modulePhiStart + (double) (imod+1) * m_moduleStepPhi;
730  if(phi_world >=min_phi && phi_world < max_phi)
731  {
732  readout_mod = imod;
733  break;
734  }
735  }
736  if(readout_mod >= m_nTpcModulesPerLayer)
737  {
738  cout << PHWHERE << "Error: readout_mod is undefined, do nothing phi_world = " << phi_world << endl;
739  return Fun4AllReturnCodes::ABORTEVENT;
740  }
741 
742  TrkrDefs::hitsetkey hitset_key = TpcDefs::genHitSetKey(layer, readout_mod, side);
743  if(m_verbosity > 3)
744  if(layer == 30)
745  cout << " world = " << world[0] << " " << world[1] << " " << world[2] << " phi_world " << phi_world*180/3.14159 << " layer " << layer << " readout_mod " << readout_mod << " side " << side << " hitsetkey " << hitset_key<< endl;
746 
747  return hitset_key;
