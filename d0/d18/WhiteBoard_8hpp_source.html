54  // type-erased value holder for move-constructible types
55  struct IHolder {
56  virtual ~IHolder() = default;
57  virtual const std::type_info& type() const = 0;

59  template <typename T,
60  typename =

62  struct HolderT : public IHolder {
63  T value;

65  HolderT(T&& v) : value(std::move(v)) {}
66  const std::type_info& type() const { return typeid(T); }

80 template <typename T>
81 inline void FW::WhiteBoard::add(const std::string& name, T&& object) {
82  if (name.empty()) {
83  throw std::invalid_argument("Object can not have an empty name");
84  }
85  if (0 < m_store.count(name)) {
86  throw std::invalid_argument("Object '" + name + "' already exists");
87  }
88  m_store.emplace(name, std::make_unique<HolderT<T>>(std::forward<T>(object)));
89  ACTS_VERBOSE("Added object '" << name << "'");
90 }

92 template <typename T>
93 inline const T& FW::WhiteBoard::get(const std::string& name) const {
94  auto it = m_store.find(name);
95  if (it == m_store.end()) {
96  throw std::out_of_range("Object '" + name + "' does not exists");
97  }
98  const IHolder* holder = it->second.get();
99  if (typeid(T) != holder->type()) {
100  throw std::out_of_range("Type missmatch for object '" + name + "'");
101  }
102  ACTS_VERBOSE("Retrieved object '" << name << "'");
103  return reinterpret_cast<const HolderT<T>*>(holder)->value;
104 }
